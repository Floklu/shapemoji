148  void DoRebuilds()
149  {
150  // Handle text objects the require an update either as a result of scale changes or legacy animation.
151  for (int i = 0; i < m_InternalUpdateQueue.Count; i++)
152  {
153  m_InternalUpdateQueue[i].InternalUpdate();
154  }
155 
156  // Handle Layout Rebuild Phase
157  for (int i = 0; i < m_LayoutRebuildQueue.Count; i++)
158  {
159  m_LayoutRebuildQueue[i].Rebuild(CanvasUpdate.Prelayout);
160  }
161 
162  if (m_LayoutRebuildQueue.Count > 0)
163  {
164  m_LayoutRebuildQueue.Clear();
165  m_LayoutQueueLookup.Clear();
166  }
167 
168  // Handle Graphic Rebuild Phase
169  for (int i = 0; i < m_GraphicRebuildQueue.Count; i++)
170  {
171  m_GraphicRebuildQueue[i].Rebuild(CanvasUpdate.PreRender);
172  }
173 
174  // If there are no objects in the queue, we don't need to clear the lists again.
175  if (m_GraphicRebuildQueue.Count > 0)
176  {
177  m_GraphicRebuildQueue.Clear();
178  m_GraphicQueueLookup.Clear();
179  }
180 
181  // Handle Culling Update
182  for (int i = 0; i < m_CullingUpdateQueue.Count; i++)
183  {
184  m_CullingUpdateQueue[i].UpdateCulling();
185  }
186 
187  // If there are no objects in the queue, we don't need to clear the lists again.
188  if (m_CullingUpdateQueue.Count > 0)
189  {
190  m_CullingUpdateQueue.Clear();
