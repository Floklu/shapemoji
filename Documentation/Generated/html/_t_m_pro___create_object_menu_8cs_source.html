200  private static void SetPositionVisibleinSceneView(RectTransform canvasRTransform, RectTransform itemTransform)
201  {
202  // Find the best scene view
203  SceneView sceneView = SceneView.lastActiveSceneView;
204  if (sceneView == null && SceneView.sceneViews.Count > 0)
205  sceneView = SceneView.sceneViews[0] as SceneView;
206 
207  // Couldn't find a SceneView. Don't set position.
208  if (sceneView == null || sceneView.camera == null)
209  return;
210 
211  // Create world space Plane from canvas position.
212  Camera camera = sceneView.camera;
213  Vector3 position = Vector3.zero;
214  Vector2 localPlanePosition;
215 
216  if (RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRTransform, new Vector2(camera.pixelWidth / 2, camera.pixelHeight / 2), camera, out localPlanePosition))
217  {
218  // Adjust for canvas pivot
219  localPlanePosition.x = localPlanePosition.x + canvasRTransform.sizeDelta.x * canvasRTransform.pivot.x;
220  localPlanePosition.y = localPlanePosition.y + canvasRTransform.sizeDelta.y * canvasRTransform.pivot.y;
221 
222  localPlanePosition.x = Mathf.Clamp(localPlanePosition.x, 0, canvasRTransform.sizeDelta.x);
223  localPlanePosition.y = Mathf.Clamp(localPlanePosition.y, 0, canvasRTransform.sizeDelta.y);
224 
225  // Adjust for anchoring
226  position.x = localPlanePosition.x - canvasRTransform.sizeDelta.x * itemTransform.anchorMin.x;
227  position.y = localPlanePosition.y - canvasRTransform.sizeDelta.y * itemTransform.anchorMin.y;
228 
229  Vector3 minLocalPosition;
230  minLocalPosition.x = canvasRTransform.sizeDelta.x * (0 - canvasRTransform.pivot.x) + itemTransform.sizeDelta.x * itemTransform.pivot.x;
231  minLocalPosition.y = canvasRTransform.sizeDelta.y * (0 - canvasRTransform.pivot.y) + itemTransform.sizeDelta.y * itemTransform.pivot.y;
232 
233  Vector3 maxLocalPosition;
234  maxLocalPosition.x = canvasRTransform.sizeDelta.x * (1 - canvasRTransform.pivot.x) - itemTransform.sizeDelta.x * itemTransform.pivot.x;
235  maxLocalPosition.y = canvasRTransform.sizeDelta.y * (1 - canvasRTransform.pivot.y) - itemTransform.sizeDelta.y * itemTransform.pivot.y;
236 
237  position.x = Mathf.Clamp(position.x, minLocalPosition.x, maxLocalPosition.x);
238  position.y = Mathf.Clamp(position.y, minLocalPosition.y, maxLocalPosition.y);
239  }
240 
241  itemTransform.anchoredPosition = position;
242  itemTransform.localRotation = Quaternion.identity;
243  itemTransform.localScale = Vector3.one;
244  }

247  private static void PlaceUIElementRoot(GameObject element, MenuCommand menuCommand)
248  {
249  GameObject parent = menuCommand.context as GameObject;
250  bool explicitParentChoice = true;
251  if (parent == null)
252  {
253  parent = GetOrCreateCanvasGameObject();
254  explicitParentChoice = false;
255 
256  // If in Prefab Mode, Canvas has to be part of Prefab contents,
257  // otherwise use Prefab root instead.
258  PrefabStage prefabStage = PrefabStageUtility.GetCurrentPrefabStage();
259  if (prefabStage != null && !prefabStage.IsPartOfPrefabContents(parent))
260  parent = prefabStage.prefabContentsRoot;
261  }
262  if (parent.GetComponentInParent<Canvas>() == null)
263  {
264  // Create canvas under context GameObject,
265  // and make that be the parent which UI element is added under.
266  GameObject canvas = CreateNewUI();
267  canvas.transform.SetParent(parent.transform, false);
268  parent = canvas;
269  }
270 
271  // Setting the element to be a child of an element already in the scene should
272  // be sufficient to also move the element to that scene.
273  // However, it seems the element needs to be already in its destination scene when the
274  // RegisterCreatedObjectUndo is performed; otherwise the scene it was created in is dirtied.
275  SceneManager.MoveGameObjectToScene(element, parent.scene);
276 
277  if (element.transform.parent == null)
278  {
279  Undo.SetTransformParent(element.transform, parent.transform, "Parent " + element.name);
280  }
281 
282  GameObjectUtility.EnsureUniqueNameForSibling(element);
283 
284  // We have to fix up the undo name since the name of the object was only known after reparenting it.
285  Undo.SetCurrentGroupName("Create " + element.name);
286 
287  GameObjectUtility.SetParentAndAlign(element, parent);
288  if (!explicitParentChoice) // not a context click, so center in sceneview
289  SetPositionVisibleinSceneView(parent.GetComponent<RectTransform>(), element.GetComponent<RectTransform>());
290 
291  Undo.RegisterCreatedObjectUndo(element, "Create " + element.name);
292 
293  Selection.activeGameObject = element;
294  }

355  // Helper function that returns a Canvas GameObject; preferably a parent of the selection, or other existing Canvas.
356  public static GameObject GetOrCreateCanvasGameObject()
357  {
358  GameObject selectedGo = Selection.activeGameObject;
359 
360  // Try to find a gameobject that is the selected GO or one if its parents.
361  Canvas canvas = (selectedGo != null) ? selectedGo.GetComponentInParent<Canvas>() : null;
362  if (IsValidCanvas(canvas))
363  return canvas.gameObject;
364 
365  // No canvas in selection or its parents? Then use any valid canvas.
366  // We have to find all loaded Canvases, not just the ones in main scenes.
367  Canvas[] canvasArray = StageUtility.GetCurrentStageHandle().FindComponentsOfType<Canvas>();
368  for (int i = 0; i < canvasArray.Length; i++)
369  if (IsValidCanvas(canvasArray[i]))
370  return canvasArray[i].gameObject;
371 
372  // No canvas in the scene at all? Then create a new one.
373  return CreateNewUI();
374  }

376  static bool IsValidCanvas(Canvas canvas)
377  {
378  if (canvas == null || !canvas.gameObject.activeInHierarchy)
379  return false;
380 
381  // It's important that the non-editable canvas from a prefab scene won't be rejected,
382  // but canvases not visible in the Hierarchy at all do. Don't check for HideAndDontSave.
383  if (EditorUtility.IsPersistent(canvas) || (canvas.hideFlags & HideFlags.HideInHierarchy) != 0)
384  return false;
385 
386  if (StageUtility.GetStageHandle(canvas.gameObject) != StageUtility.GetCurrentStageHandle())
387  return false;
388 
389  return true;
390  }
