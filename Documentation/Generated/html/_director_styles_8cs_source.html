332  public static string Elipsify(string label, float destinationWidth, float neededWidth)
333  {
334  var ret = label;
335 
336  if (label.Length == 0)
337  return ret;
338 
339  if (destinationWidth < neededWidth)
340  {
341  float averageWidthOfOneChar = neededWidth / label.Length;
342  int floor = Mathf.Max((int)Mathf.Floor(destinationWidth / averageWidthOfOneChar), 0);
343 
344  if (floor < k_Elipsis.Length)
345  ret = string.Empty;
346  else if (floor == k_Elipsis.Length)
347  ret = k_Elipsis;
348  else if (floor < label.Length)
349  ret = label.Substring(0, floor - k_Elipsis.Length) + k_Elipsis;
350  }
351 
352  return ret;
353  }
