286

289  m_ProgressPercentage = (float)m_ScanningCurrentFileIndex / m_ScanningTotalFiles;

292  if (ShouldIgnoreFile(assetFilePath))

303

338  static void ScanProjectFileAsync(AssetFileRecord fileRecord)
339  {
340  if (m_CancelScanProcess)
341  return;
342 
343  // Read the asset data file
344  string assetDataFile = string.Empty;
345  bool hasFileChanged = false;
346 
347  try
348  {
349  assetDataFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetFilePath);
350  }
351  catch
352  {
353  // Continue to the next asset if we can't read the current one.
354  return;
355  }
356 
357  // Read the asset meta data file
358  string assetMetaFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetMetaFilePath);
359  bool hasMetaFileChanges = false;
360 
361  foreach (AssetConversionRecord record in m_ConversionData.assetRecords)
362  {
363  if (assetDataFile.Contains(record.target))
364  {
365  hasFileChanged = true;
366 
367  assetDataFile = assetDataFile.Replace(record.target, record.replacement);
368  }
369 
371  if (assetMetaFile.Contains(record.target))
372  {
373  hasMetaFileChanges = true;
374 
375  assetMetaFile = assetMetaFile.Replace(record.target, record.replacement);
376  }
377  }
378 
379  if (hasFileChanged)
380  {
381  AssetModificationRecord modifiedAsset;
382  modifiedAsset.assetFilePath = fileRecord.assetFilePath;
383  modifiedAsset.assetDataFile = assetDataFile;
384 
385  m_ModifiedAssetList.Add(modifiedAsset);
386 
387  m_ProjectScanResults += fileRecord.assetFilePath + "\n";
388  }
389 
390  if (hasMetaFileChanges)
391  {
392  AssetModificationRecord modifiedAsset;
393  modifiedAsset.assetFilePath = fileRecord.assetMetaFilePath;
394  modifiedAsset.assetDataFile = assetMetaFile;
395 
396  m_ModifiedAssetList.Add(modifiedAsset);
397 
398  m_ProjectScanResults += fileRecord.assetMetaFilePath + "\n";
399  }
