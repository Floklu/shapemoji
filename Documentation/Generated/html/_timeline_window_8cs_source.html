307  void DetectStateChanges()
308  {
309  if (state != null)
310  {
311  state.editSequence.ResetIsReadOnly(); //Force reset readonly for asset flag for each frame.
312  // detect if the sequence was removed under our feet
313  if (m_LastFrameHadSequence && state.editSequence.asset == null)
314  {
315  ClearCurrentTimeline();
316  }
317  m_LastFrameHadSequence = state.editSequence.asset != null;
318 
319  // the currentDirector can get set to null by a deletion or scene unloading so polling is required
320  if (state.editSequence.director == null)
321  {
322  state.recording = false;
323  state.previewMode = false;
324 
325  //Case 1201405 : Check if the lock state is valid with the lock tracker state
326  if (locked != m_LockTracker.isLocked)
327  m_LockTracker.isLocked = locked;
328 
329  if (!locked && m_LastFrameHadSequence)
330  {
331  // the user may be adding a new PlayableDirector to a selected GameObject, make sure the timeline editor is shows the proper director if none is already showing
332  var selectedGameObject = Selection.activeObject != null ? Selection.activeObject as GameObject : null;
333  var selectedDirector = selectedGameObject != null ? selectedGameObject.GetComponent<PlayableDirector>() : null;
334  if (selectedDirector != null)
335  {
336  SetCurrentTimeline(selectedDirector);
337  }
338  }
339  }
340  else
341  {
342  // the user may have changed the timeline associated with the current director
343  if (state.editSequence.asset != state.editSequence.director.playableAsset)
344  {
345  if (!locked)
346  {
347  SetCurrentTimeline(state.editSequence.director);
348  }
349  else
350  {
351  // Keep locked on the current timeline but set the current director to null since it's not the timeline owner anymore

420  public void RebuildGraphIfNecessary(bool evaluate = true)
421  {
422  if (state == null || state.editSequence.director == null || state.editSequence.asset == null)
423  return;
424

426  {

429

441  state.editSequence.time = time;
442

445
